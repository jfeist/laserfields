63  function interpolate_array(x_old,f_old,x_new,degree) result(f_new)
64  real(dp), dimension(:), intent(in) :: x_old, f_old, x_new
65  real(dp), dimension(size(x_new)) :: f_new
66 
67  integer, intent(in) :: degree
68  integer :: ii, npoints_old, npoints_new, nearest_node, spoint, npoints_int
69 
70  if (size(x_old)/=size(f_old)) stop 'ERROR: number of nodes and function values do not agree in interpolate!'
71 
72  npoints_new = size(x_new)
73  npoints_old = size(x_old)
74  ! number of points used for interpolation
75  npoints_int=degree+1
76 
77  do ii = 1, npoints_new
78  if (x_new(ii) < x_old(1) .or. x_new(ii) > x_old(npoints_old)) then
79  stop 'ERROR: extrapolation would be necessary in interpolate!'
80  end if
81 
82  ! get nearest node smaller then x_new(ii)
83  nearest_node = binary_search(x_old,x_new(ii))
84  ! get starting point for interpolation
85  spoint = min(max(nearest_node-(npoints_int-1)/2,1),npoints_old+1-npoints_int)
86  ! call interpolation routine
87  f_new(ii) = lagrange_interpol(x_old(spoint:spoint+degree),f_old(spoint:spoint+degree),npoints_int,x_new(ii))
88  end do
89  end function interpolate_array

92  real(dp) pure function lagrange_interpol(xk,yk,n,x) result(y)
93 
94  integer, intent(in) :: n
95 
96  real(dp), intent(in) :: xk(n)
97 
98  real(dp), intent(in) :: yk(n)
99 
100  real(dp), intent(in) :: x
101  real(dp) :: wk(n), nom, denom, diff
102  integer :: ii, jj
103  do jj = 1, n
104  wk(jj) = 1
105  do ii = 1, n
106  if (ii/=jj) wk(jj) = wk(jj) / (xk(jj)-xk(ii))
107  end do
108  end do
109  nom = 0
110  denom = 0
111  do jj = 1, n
112  diff = abs(x-xk(jj))
113  ! this guards against x=xk(jj)=0
114  if (diff/=0) diff = 2*diff / (abs(x)+abs(xk(jj)))
115  if (diff < 1.e-12) then
116  y = yk(jj)
117  return
118  end if
119  nom = nom + wk(jj)*yk(jj) / (x - xk(jj))
120  denom = denom + wk(jj) / (x - xk(jj))
121  end do
122  y = nom/denom
123  end function lagrange_interpol

126  pure integer function binary_search(xs,x) result(il)
127 
128  real(dp), intent(in) :: xs(:)
129 
130  real(dp), intent(in) :: x
131  integer :: n, im, iu
132  logical :: increasing
133 
134  n = size(xs)
135  ! we do not check whether the sequence is actually monotonic, but we handle increasing and decreasing sequences
136  increasing = xs(1) < xs(n)
137 
138  if (increasing) then
139  if (x < xs(1)) then; il = 1; return
140  else if (x > xs(n)) then; il = n; return
141  end if
142  else
143  if (x > xs(1)) then; il = 1; return
144  else if (x < xs(n)) then; il = n; return
145  end if
146  end if
147 
148  il=0 ! initial value for lower border
149  iu=n+1 ! initial value for upper border
150  do while (iu-il>1)
151  im=(iu+il)/2
152  if (increasing.eqv.(x > xs(im))) then
153  il=im
154  else
155  iu=im
156  end if
157  end do
158  end function binary_search
