8  integer :: ii, i_field, nsteps
9  real(dp) :: omega, de, emax

15  parfile = 'laserfields.in'
16  if (command_argument_count() >= 1) then
17  call get_command_argument(1,parfile)
18  end if
19  call laserfields_read_parameters(trim(parfile))
20 
21  if (.not.laserfields_can_get_fourier()) then
22  write(0,*) 'ERROR: can not get analytical fourier transform for these laser fields. Stopping!'
23  stop 733
24  end if

26  allocate(lf_ft(0:n_laserfields))
27 
28  ! determine energy resolution by estimating fourier bandwidth
29  de = 1.d0
30  emax = 0.d0
31  do i_field = 1, n_laserfields
32  de = min(de,twopi/(100*all_laserfields(i_field)%duration))
33  ! go up to twice the maximum frequency - for really broadband pulses, this might not actually be enough
34  emax = max(emax,all_laserfields(i_field)%omega*2.01) ! use 2.01 here so we do not get omega==lf%omega as an argument, where some of the routines have problems
35  end do
36  nsteps = nint(emax/de)
37  de = emax/(nsteps+1)
38 
39  allocate(eabs(nsteps), ephase(nsteps))
40  do ii = 1, nsteps
41  omega = ii * de
42  do i_field = 1, n_laserfields
43  lf_ft(i_field) = get_el_fourier_transform(all_laserfields(i_field), omega)
44  end do
45  lf_ft(0) = sum(lf_ft(1:))
46  alft = get_al_fourier_transform(omega)
47 
48  eabs(ii) = abs(lf_ft(0))
49  ephase(ii) = atan2(aimag(lf_ft(0)),real(lf_ft(0)))
50  ! we are not unrolling the phases here (i.e. removing phase jumps by 2 pi)
51  ! neither are we shifting in time to get a flatter phase overall
52  ! write(6,'(9999(1x,g22.14e3))') omega, lf_ft(0), alft, lf_ft(1:)
53  end do
54 
55  call unroll_phase(ephase)
56  do ii = 1, nsteps
57  omega = ii * de
58  write(6,'(9999(1x,g22.14e3))') omega, eabs(ii), ephase(ii)
59  end do

63  subroutine unroll_phase(phases)
64  ! takes an array of phases and 'unrolls' it, i.e. removes phase jumps by two pi
65  ! this is the same as saying phi_new(x) = int_0^x dphi/dx dx,
66  ! where dphi/dx is the local derivative and has no jumps
67  real(dp), dimension(:), intent(inout) :: phases
68  real(dp) :: phase_diff, acc_phase
69  integer :: ii
70 
71  ! acc_phase keeps track of how many phase jumps by TWOPI we have accumulated
72  acc_phase = 0.d0
73  do ii = 2, size(phases)
74  ! the phase difference between the previous and current step,
75  ! including a possible phase jump of +-TWOPI
76  phase_diff = phases(ii) - phases(ii-1) - acc_phase
77 
78  ! add small guard value to not change phase jumps by pi
79  if (abs(phase_diff-twopi)<abs(phase_diff)-1.d-3) then
80  ! if +TWOPI
81  acc_phase = acc_phase + twopi
82  phase_diff = phase_diff - twopi
83  else if (abs(phase_diff+twopi)<abs(phase_diff)-1.d-3) then
84  ! if -TWOPI
85  acc_phase = acc_phase - twopi
86  phase_diff = phase_diff + twopi
87  end if
88  phases(ii) = phases(ii-1) + phase_diff
89  end do
90  end subroutine unroll_phase
91  !---------------------------------------------------------------------------
92 end program plotlaserfourier
