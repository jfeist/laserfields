27 /* Available at: http://ab-initio.mit.edu/Faddeeva
28 
29  Computes various error functions (erf, erfc, erfi, erfcx),
30  including the Dawson integral, in the complex plane, based
31  on algorithms for the computation of the Faddeeva function
32  w(z) = exp(-z^2) * erfc(-i*z).
33  Given w(z), the error functions are mostly straightforward
34  to compute, except for certain regions where we have to
35  switch to Taylor expansions to avoid cancellation errors
36  [e.g. near the origin for erf(z)].
37 
38  To compute the Faddeeva function, we use a combination of two
39  algorithms:
40 
41  For sufficiently large |z|, we use a continued-fraction expansion
42  for w(z) similar to those described in:
43 
44  Walter Gautschi, "Efficient computation of the complex error
45  function," SIAM J. Numer. Anal. 7(1), pp. 187-198 (1970)
46 
47  G. P. M. Poppe and C. M. J. Wijers, "More efficient computation
48  of the complex error function," ACM Trans. Math. Soft. 16(1),
49  pp. 38-46 (1990).
50 
51  Unlike those papers, however, we switch to a completely different
52  algorithm for smaller |z|:
53 
54  Mofreh R. Zaghloul and Ahmed N. Ali, "Algorithm 916: Computing the
55  Faddeyeva and Voigt Functions," ACM Trans. Math. Soft. 38(2), 15
56  (2011).
57 
58  (I initially used this algorithm for all z, but it turned out to be
59  significantly slower than the continued-fraction expansion for
60  larger |z|. On the other hand, it is competitive for smaller |z|,
61  and is significantly more accurate than the Poppe & Wijers code
62  in some regions, e.g. in the vicinity of z=1+1i.)
63 
64  Note that this is an INDEPENDENT RE-IMPLEMENTATION of these algorithms,
65  based on the description in the papers ONLY. In particular, I did
66  not refer to the authors' Fortran or Matlab implementations, respectively,
67  (which are under restrictive ACM copyright terms and therefore unusable
68  in free/open-source software).
69 
70  Steven G. Johnson, Massachusetts Institute of Technology
71  http://math.mit.edu/~stevenj
72  October 2012.

75  or rather the scaled function erfcx(x) = exp(x*x)*erfc(x),
76  is supplied for REAL arguments x. I originally used an
77  erfcx routine derived from DERFC in SLATEC, but I have
78  since replaced it with a much faster routine written by
79  me which uses a combination of continued-fraction expansions
80  and a lookup table of Chebyshev polynomials. For speed,
81  I implemented a similar algorithm for Im[w(x)] of real x,
82  since this comes up frequently in the other error functions.

245 // compute the error function erf(z)
246 cmplx FADDEEVA(erf)(cmplx z, double relerr)
247 {
248  double x = creal(z), y = cimag(z);
249 
250  if (y == 0)
251  return CMPLX(FADDEEVA_RE(erf)(x),
252  y); // preserve sign of 0
253  if (x == 0) // handle separately for speed & handling of y = Inf or NaN
254  return CMPLX(x, // preserve sign of 0
255  /* handle y -> Inf limit manually, since
256  exp(y^2) -> Inf but Im[w(y)] -> 0, so
257  IEEE will give us a NaN when it should be Inf */
258  y*y > 720 ? (y > 0 ? Inf : -Inf)
259  : exp(y*y) * FADDEEVA(w_im)(y));
260 
261  double mRe_z2 = (y - x) * (x + y); // Re(-z^2), being careful of overflow
262  double mIm_z2 = -2*x*y; // Im(-z^2)
263  if (mRe_z2 < -750) // underflow
264  return (x >= 0 ? 1.0 : -1.0);
265 
266  /* Handle positive and negative x via different formulas,
267  using the mirror symmetries of w, to avoid overflow/underflow
268  problems from multiplying exponentially large and small quantities. */
269  if (x >= 0) {
270  if (x < 5e-3) {
271  if (fabs(y) < 5e-3)
272  goto taylor;
273  else if (fabs(mIm_z2) < 5e-3)
274  goto taylor_erfi;
275  }
276  /* don't use complex exp function, since that will produce spurious NaN
277  values when multiplying w in an overflow situation. */
278  return 1.0 - exp(mRe_z2) *
279  (CMPLX(cos(mIm_z2), sin(mIm_z2))
280  * FADDEEVA(w)(CMPLX(-y,x), relerr));
281  }
282  else { // x < 0
283  if (x > -5e-3) { // duplicate from above to avoid fabs(x) call
284  if (fabs(y) < 5e-3)
285  goto taylor;
286  else if (fabs(mIm_z2) < 5e-3)
287  goto taylor_erfi;
288  }
289  else if (isnan(x))
290  return CMPLX(NaN, y == 0 ? 0 : NaN);
291  /* don't use complex exp function, since that will produce spurious NaN
292  values when multiplying w in an overflow situation. */
293  return exp(mRe_z2) *
294  (CMPLX(cos(mIm_z2), sin(mIm_z2))
295  * FADDEEVA(w)(CMPLX(y,-x), relerr)) - 1.0;
296  }
297 
298  // Use Taylor series for small |z|, to avoid cancellation inaccuracy
299  // erf(z) = 2/sqrt(pi) * z * (1 - z^2/3 + z^4/10 - ...)
300  taylor:
301  {
302  cmplx mz2 = CMPLX(mRe_z2, mIm_z2); // -z^2
303  return z * (1.1283791670955125739
304  + mz2 * (0.37612638903183752464
305  + mz2 * 0.11283791670955125739));
306  }
307 
308  /* for small |x| and small |xy|,
309  use Taylor series to avoid cancellation inaccuracy:
310  erf(x+iy) = erf(iy)
311  + 2*exp(y^2)/sqrt(pi) *
312  [ x * (1 - x^2 * (1+2y^2)/3 + x^4 * (3+12y^2+4y^4)/30 + ...
313  - i * x^2 * y * (1 - x^2 * (3+2y^2)/6 + ...) ]
314  where:
315  erf(iy) = exp(y^2) * Im[w(y)]
316  */
317  taylor_erfi:
318  {
319  double x2 = x*x, y2 = y*y;
320  double expy2 = exp(y2);
321  return CMPLX
322  (expy2 * x * (1.1283791670955125739
323  - x2 * (0.37612638903183752464
324  + 0.75225277806367504925*y2)
325  + x2*x2 * (0.11283791670955125739
326  + y2 * (0.45135166683820502956
327  + 0.15045055561273500986*y2))),
328  expy2 * (FADDEEVA(w_im)(y)
329  - x2*y * (1.1283791670955125739
330  - x2 * (0.56418958354775628695
331  + 0.37612638903183752464*y2))));
332  }
333 }

364 // erfc(z) = 1 - erf(z)
365 cmplx FADDEEVA(erfc)(cmplx z, double relerr)
366 {
367  double x = creal(z), y = cimag(z);
368 
369  if (x == 0.)
370  return CMPLX(1,
371  /* handle y -> Inf limit manually, since
372  exp(y^2) -> Inf but Im[w(y)] -> 0, so
373  IEEE will give us a NaN when it should be Inf */
374  y*y > 720 ? (y > 0 ? -Inf : Inf)
375  : -exp(y*y) * FADDEEVA(w_im)(y));
376  if (y == 0.) {
377  if (x*x > 750) // underflow
378  return CMPLX(x >= 0 ? 0.0 : 2.0,
379  -y); // preserve sign of 0
380  return CMPLX(x >= 0 ? exp(-x*x) * FADDEEVA_RE(erfcx)(x)
381  : 2. - exp(-x*x) * FADDEEVA_RE(erfcx)(-x),
382  -y); // preserve sign of zero
383  }
384 
385  double mRe_z2 = (y - x) * (x + y); // Re(-z^2), being careful of overflow
386  double mIm_z2 = -2*x*y; // Im(-z^2)
387  if (mRe_z2 < -750) // underflow
388  return (x >= 0 ? 0.0 : 2.0);
389 
390  if (x >= 0)
391  return cexp(CMPLX(mRe_z2, mIm_z2))
392  * FADDEEVA(w)(CMPLX(-y,x), relerr);
393  else
394  return 2.0 - cexp(CMPLX(mRe_z2, mIm_z2))
395  * FADDEEVA(w)(CMPLX(y,-x), relerr);
396 }

459  // Use Taylor series for small |z|, to avoid cancellation inaccuracy
460  // dawson(z) = z - 2/3 z^3 + 4/15 z^5 + ...
461  taylor:
462  return z * (1.
463  + mz2 * (0.6666666666666666666666666666666666666667
464  + mz2 * 0.2666666666666666666666666666666666666667));
465 
466  /* for small |y| and small |xy|,
467  use Taylor series to avoid cancellation inaccuracy:
468  dawson(x + iy)
469  = D + y^2 (D + x - 2Dx^2)
470  + y^4 (D/2 + 5x/6 - 2Dx^2 - x^3/3 + 2Dx^4/3)
471  + iy [ (1-2Dx) + 2/3 y^2 (1 - 3Dx - x^2 + 2Dx^3)
472  + y^4/15 (4 - 15Dx - 9x^2 + 20Dx^3 + 2x^4 - 4Dx^5) ] + ...
473  where D = dawson(x)
474 
475  However, for large |x|, 2Dx -> 1 which gives cancellation problems in
476  this series (many of the leading terms cancel). So, for large |x|,
477  we need to substitute a continued-fraction expansion for D.
478 
479  dawson(x) = 0.5 / (x-0.5/(x-1/(x-1.5/(x-2/(x-2.5/(x...))))))
480 
481  The 6 terms shown here seems to be the minimum needed to be
482  accurate as soon as the simpler Taylor expansion above starts
483  breaking down. Using this 6-term expansion, factoring out the
484  denominator, and simplifying with Maple, we obtain:
485 
486  Re dawson(x + iy) * (-15 + 90x^2 - 60x^4 + 8x^6) / x
487  = 33 - 28x^2 + 4x^4 + y^2 (18 - 4x^2) + 4 y^4
488  Im dawson(x + iy) * (-15 + 90x^2 - 60x^4 + 8x^6) / y
489  = -15 + 24x^2 - 4x^4 + 2/3 y^2 (6x^2 - 15) - 4 y^4
490 
491  Finally, for |x| > 5e7, we can use a simpler 1-term continued-fraction
492  expansion for the real part, and a 2-term expansion for the imaginary
493  part. (This avoids overflow problems for huge |x|.) This yields:
494 
495  Re dawson(x + iy) = [1 + y^2 (1 + y^2/2 - (xy)^2/3)] / (2x)
496  Im dawson(x + iy) = y [ -1 - 2/3 y^2 + y^4/15 (2x^2 - 4) ] / (2x^2 - 1)
497 
498  */
499  taylor_realaxis:
500  {
501  double x2 = x*x;
502  if (x2 > 1600) { // |x| > 40
503  double y2 = y*y;
504  if (x2 > 25e14) {// |x| > 5e7
505  double xy2 = (x*y)*(x*y);
506  return CMPLX((0.5 + y2 * (0.5 + 0.25*y2
507  - 0.16666666666666666667*xy2)) / x,
508  y * (-1 + y2 * (-0.66666666666666666667
509  + 0.13333333333333333333*xy2
510  - 0.26666666666666666667*y2))
511  / (2*x2 - 1));
512  }
513  return (1. / (-15 + x2*(90 + x2*(-60 + 8*x2)))) *
514  CMPLX(x * (33 + x2 * (-28 + 4*x2)
515  + y2 * (18 - 4*x2 + 4*y2)),
516  y * (-15 + x2 * (24 - 4*x2)
517  + y2 * (4*x2 - 10 - 4*y2)));
518  }
519  else {
520  double D = spi2 * FADDEEVA(w_im)(x);
521  double x2 = x*x, y2 = y*y;
522  return CMPLX
523  (D + y2 * (D + x - 2*D*x2)
524  + y2*y2 * (D * (0.5 - x2 * (2 - 0.66666666666666666667*x2))
525  + x * (0.83333333333333333333
