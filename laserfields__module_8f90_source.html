235  subroutine read_laserfield_from_file(lf)
236  use laserfields_miscfuncs
237  use atomic_units
238  type(laserfield), intent(inout) :: lf
239  integer :: ii, io_error, unit, npoints
240  character(len=200) :: tmpstr
241  real(dp), dimension(:), allocatable :: tmptt, tmpvals
242  real(dp) :: dt, lastzerocrossing, starttime, endtime
243 
244  write(6,*) '# Reading laserfield from file: ', trim(lf%datafile)
245 
246  unit = get_unused_unit()
247  open(unit,file=trim(lf%datafile),status='old',action='read',iostat=io_error)
248  if (io_error /= 0) then
249  write(0,*) 'ERROR opening file for laserfield, filename =', trim(lf%datafile)
250  stop 511
251  end if
252 
253  npoints=0
254  ! count number of valid lines in file
255  do while (io_error==0) ! read lines until an IO error ocurs, i.e. (normally) EOF
256  read(unit,'(a200)',iostat=io_error) tmpstr
257  tmpstr = adjustl(tmpstr)
258  ! at a comment, empty line or end of file, cycle the loop
259  if (tmpstr(1:1)=='#' .or. tmpstr=='' .or. io_error/=0) cycle
260  ! otherwise, the line has data
261  npoints=npoints+1
262  end do
263 
264  if (npoints == 0) then
265  write(0,*) 'ERROR: No data points found in file for laserfield. Stop.'
266  stop 512
267  end if
268 
269  write(6,*) '# Number of data points found:', npoints
270  allocate(tmptt(npoints), tmpvals(npoints))
271 
272  ! now read in
273  ii = 0
274  rewind(unit, iostat=io_error)
275  do while (io_error==0)
276  read(unit,'(a200)',iostat=io_error) tmpstr
277  tmpstr = adjustl(tmpstr)
278  if (tmpstr(1:1)=='#' .or. tmpstr=='' .or. io_error/=0) cycle
279 
280  ii = ii + 1
281  read(tmpstr,*) tmptt(ii), tmpvals(ii)
282  end do
283  close(unit)
284 
285  !**************
286  ! analyze the data we have read to guess some information about the field
287  starttime = tmptt(1)
288  endtime = tmptt(npoints)
289 
290  lf%TX = huge(1.d0)
291  dt = huge(1.d0)
292  lastzerocrossing = starttime
293  do ii = 2, npoints
294  if (tmptt(ii)<=tmptt(ii-1)) then
295  write(0,*) 'ERROR: times in data file for laser field must be monotonously increasing! filename =', trim(lf%datafile)
296  stop 514
297  end if
298 
299  dt = min(dt,tmptt(ii)-tmptt(ii-1))
300  if (sign(1.d0,tmpvals(ii))/=sign(1.d0,tmpvals(ii-1))) then
301  ! we have crossed a zero, estimate TX with this
302  lf%TX = min(lf%TX,2*(tmptt(ii)-lastzerocrossing))
303  lastzerocrossing = tmptt(ii)
304  end if
305  end do
306 
307  ! ensure small enough dt to allow for good interpolation and numeric integration/differentiation
308  ! note that TX will usually be much smaller than the "real" TX already, so this should
309  ! be small enough by far
310  dt = min(dt,lf%TX/200)
311 
312  !**************
313  ! transform the field with arbitrary times to a grid with equal spacing for the time steps
314  ! watch out, we are reusing npoints
315  npoints = nint((endtime - starttime) / dt) + 1 ! the "+ 1" is for the endpoint
316  ! this is the actual dt we use to have exactly npoints from starttime to endtime
317  dt = (endtime - starttime) / (npoints-1)
318  allocate(lf%tt(npoints), lf%EE(npoints), lf%AA(npoints), lf%ZZ(npoints))
319  do ii = 1, npoints
320  lf%tt(ii) = starttime + (ii-1) * dt
321  end do
322 
323  ! Interpolate to get laser field on regularly spaced points
324  if (lf%is_vecpot) then
325  lf%AA(:) = interpolate(tmptt,tmpvals,lf%tt,degree=6)
326  call laserfield_numerical_derivation_e_from_a(lf)
327  else
328  lf%EE(:) = interpolate(tmptt,tmpvals,lf%tt,degree=6)
329  call laserfield_numerical_integration_a_from_e(lf)
330  end if
331  call laserfield_numerical_integration_z_from_a(lf)
332 
333  ! set guessed parameters
334  lf%E0 = maxval(abs(lf%EE))
335  lf%intensity_Wcm2 = lf%E0**2 / au_wcm2toel2
336  lf%omega = twopi / lf%TX
337  lf%lambda_nm = lf%TX / au_nm * au_c
338  lf%peak_time = (starttime + endtime) / 2
339  lf%peak_time_as = lf%peak_time / au_as
340  lf%duration = endtime - starttime
341  lf%duration_as = lf%duration / au_as
342 
343  end subroutine read_laserfield_from_file

363  subroutine laserfield_numerical_integration_a_from_e(lf)
364  type(laserfield), intent(inout) :: lf
365  integer :: ii, jj, nsteps, simpson_fac
366  real(dp) :: el, zeit
367  ! we have a laser field where we can get the electric field and
368  ! want to numerically integrate that to get the vector potential A(t) = -Int E(t) dt
369  ! we use the composite Simpson rule
370 
371  ! nsteps HAS to be ODD for Simpson rule, so that the number of intervals is even
372  nsteps = 21
373 
374  ! initialize first value to zero
375  lf%AA(1) = 0.d0
376  do ii = 2, size(lf%AA)
377  ! for each step we want to take, we do nsteps integration steps with the trapezoid rule
378  zeit = lf%tt(ii-1)
379  el = get_el(lf,zeit)
380 
381  ! do integration from tt(ii-1) to tt(ii)
382  ! Int_a^b f(t) dt ~ (f(a) + 4*f(a+h) + 2*f(a+2*h) + 4*f(a+3*h) + ... + 4*f(b-h) + f(b)) * h/3
383  ! where h = (b-a)/(nsteps-1)
384  lf%AA(ii) = -el
385  simpson_fac = 4
386  do jj = 2, nsteps-1
387  zeit = (lf%tt(ii-1) * (nsteps-jj) + lf%tt(ii) * (jj-1)) / (nsteps-1)
388  ! jj = 1 -> zeit = lf%tt(ii-1) * (nsteps-1) / (nsteps-1) -> lf%tt(ii-1) -> ok
389  ! jj = nsteps -> zeit = lf%tt(ii) * (nsteps-1) / (nsteps-1) -> lf%tt(ii) -> ok
390  el = get_el(lf,zeit)
391  ! do not forget the minus sign!
392  lf%AA(ii) = lf%AA(ii) - simpson_fac * el
393  ! 6 - 2 = 4, 6 - 4 = 2 -> alternating between 4 and 2
394  simpson_fac = 6 - simpson_fac
395  end do
396  ! last step
397  zeit = lf%tt(ii)
398  el = get_el(lf,zeit)
399  lf%AA(ii) = lf%AA(ii) - el
400  ! int = sum(...) * h/3
401  lf%AA(ii) = lf%AA(ii) * (lf%tt(ii)-lf%tt(ii-1))/(3*(nsteps-1))
402  ! A(t+dt) = A(t) - Int_t^{t+dt} E(t) dt
403  lf%AA(ii) = lf%AA(ii) + lf%AA(ii-1)
404  !write(60,'(99g25.12)') lf%tt(ii), lf%AA(ii)
405  end do
406  end subroutine laserfield_numerical_integration_a_from_e

522  ! this calculates the envelope of the laser field
523  ! as well as the first derivative (by time) of the envelope
524  ! the derivative is needed for calculating E(t) = -dA/dt if the laserfield describes A(t) (lf%is_vecpot is .true.)
525  subroutine lf_get_envelope(lf,zeit,env,envpr)
526  type(laserfield), intent(in) :: lf
527  real(dp), intent(in) :: zeit
528  real(dp), intent(out) :: env, envpr
529  real(dp) :: trel, ttmp
530 
531  trel = zeit - lf%peak_time
532 
533  select case (lf%form)
534  case('gaussianF') ! laser pulse with Gaussian envelope, lf%duration is FWHM of electric field envelope
535  env = exp(-trel**2*log(16.d0)/lf%duration**2)
536  envpr = -2*trel*log(16.d0)/lf%duration**2 * env
537  case('gaussianI') ! laser pulse with Gaussian envelope, lf%duration is FWHM of intensity envelope
538  env = exp(-trel**2*log( 4.d0)/lf%duration**2)
539  envpr = -2*trel*log( 4.d0)/lf%duration**2 * env
540  case('linear')
541  ! "linear" is a field with constant intensity for lf%duration, and linear rampon/rampoff of duration lf%rampon at the beginning and end
542  if (abs(trel) < lf%duration/2) then
543  env = 1.d0
544  envpr = 0.d0
545  else if (abs(trel) < lf%duration/2 + lf%rampon) then
546  env = 1.d0 - (abs(trel)-lf%duration/2) / lf%rampon
547  envpr = -sign(1.d0,trel) / lf%rampon
548  else
549  env = 0.d0
550  envpr = 0.d0
551  end if
552  case('linear2')
553  ! same as linear, but with sin2 rampon/rampoff
554  if (abs(trel) < lf%duration/2) then
555  env = 1.d0
556  envpr = 0.d0
557  else if (abs(trel) < lf%duration/2 + lf%rampon) then
558  ttmp = 1.d0 - (abs(trel)-lf%duration/2) / lf%rampon
559  env = sin(pio2*ttmp)**2
560  envpr = -sign(1.d0,trel) * sin(pi*trel) * pio2/lf%rampon
561  else
562  env = 0.d0
563  envpr = 0.d0
564  end if
565  case ('sin2')
566  if (abs(trel) < lf%duration/2) then
567  env = cos(pi*trel/lf%duration)**2
568  ! 2 sin(wt) cos(wt) = sin(2wt)
569  envpr = -sin(twopi*trel/lf%duration) * pi/lf%duration
570  else
571  env = 0.d0
572  envpr = 0.d0
573  end if
574  case ('sin_exp')
575  if (abs(trel) < lf%duration/2) then
576  env = cos(pi*trel/lf%duration)**lf%form_exponent
577  envpr = -sin(pi*trel/lf%duration) * lf%form_exponent * &
578  & cos(pi*trel/lf%duration)**(lf%form_exponent-1) * pi/lf%duration
579  else
580  env = 0.d0
581  envpr = 0.d0
582  end if
583  case default
584  write(0,'(2A)') 'ERROR! laser field form unknown, form = ', trim(lf%form)
585  stop 516
586  end select
587 
588  env = lf%E0 * env
589  envpr = lf%E0 * envpr
590 
591  end subroutine lf_get_envelope

593  complex(dpc) function lf_envelope_fourier(lf,omega) result(val)
594  use atomic_units
595  ! return the fourier transform of the envelope of the laser field
596  ! we write the whole pulse as
597  ! f(t) = (env(t) exp(IU*(phi0 + w0*tp + chirp*tp**2)) + c.c. ) / (2*IU), where tp = t-tpeak
598  ! for the fourier transform we include the chirp term exp(i chirp (t-tpeak)**2) in the envelope.
599  ! the fourier transform of the envelope is then a complex function.
600  ! however, for unchirped pulses, the result will be purely real!
601  type(laserfield), intent(in) :: lf
602  real(dp), intent(in) :: omega
603  real(dp) :: chirp
604  complex(dpc) :: z
605 
606  ! important - use omega, not lf%omega as the argument here!!
607 
608  ! instantaneous frequency:
609  ! w(t) = lf%omega * (1.d0 + lf%linear_chirp_rate_w0as/au_as * (zeit-lf%peak_time))
610  chirp = lf%omega * lf%linear_chirp_rate_w0as / au_as
611 
612  ! for the various calculations, see chirped_fourier.nb in the mathematica subversion directory
613  select case (lf%form)
614  case('gaussianF')
615  ! F[exp(-z*t**2)] = exp(-w**2/4*z)/sqrt(2*z) (for real(z)>0)
616  z = log(16.d0)/lf%duration**2 - iu * chirp
617  val = exp(-omega**2/(4*z)) / sqrt(2*z)
618  case('gaussianI')
619  z = log( 4.d0)/lf%duration**2 - iu * chirp
620  val = exp(-omega**2/(4*z)) / sqrt(2*z)
621  case('linear')
622  if (chirp /= 0.d0) then
623  write(0,'(A)') 'ERROR! fourier transform of "linear" envelope with chirp not implemented!'
624  stop 676
625  end if
626  val = sqrt(8.d0/pi) * sin(omega*lf%rampon/2) * sin(omega*(lf%rampon+lf%duration)/2) / (lf%rampon * omega**2)
627  case('linear2')
628  if (chirp /= 0.d0) then
629  write(0,'(A)') 'ERROR! linear2 fourier transform with chirp not implemented!'
630  stop 676
631  end if
632  val = sqrt(2*pi**3) * cos(omega*lf%rampon/2) * sin(omega*(lf%rampon+lf%duration)/2) / (pi**2*omega - lf%rampon**2*omega**3)
633  case ('sin2')
634  if (chirp == 0.d0) then ! the expression with chirp can not be evaluated with chirp == 0, so we take this as a special case
635  val = sqrt(8.d0*pi**3) * sin(omega*lf%duration/2)/(twopi**2*omega - omega**3*lf%duration**2)
636  else
637  ! now we use that cos(pi*t/T)**2 * exp(i*c*t**2) can be written as 0.5 exp(i*c*t**2) + 0.25 exp(i*c*t**2 - 2*i*pi*t/T) + 0.25 exp(i*c*t**2 + 2*i*pi*t/T)
638  ! the integral of exp(IU*(a*t+b*t**2)) from t=-T/2 to t=T/2 can be calculated analytically and is implemented in the function below
639  ! the arguments are a={-omega, -2*pi/T-omega, 2*pi/T-omega} and b=chirp
640  val = expiatbt2_intt( - omega, chirp, lf%duration)/2 &
641  & + expiatbt2_intt(-twopi/lf%duration - omega, chirp, lf%duration)/4 &
642  & + expiatbt2_intt( twopi/lf%duration - omega, chirp, lf%duration)/4
643  end if
644  case ('sin_exp')
645  stop 'ERROR! sin_exp fourier transform too complicated to implement'
646  case default
647  stop 'ERROR! unknown laser field form'
648  end select
649 
650  val = lf%E0 * val
651 
652  end function lf_envelope_fourier
653  !---------------------------------------------------------------------------
654  ! returns the result of the integral Int(exp(IU*(a*t+b*t**2)),{t,-T/2,T/2}) / sqrt(TWOPI)
655  complex(dpc) function expiatbt2_intt(a,b,T) result(res)
656  use faddeeva, ONLY : erf
657  real(dp), intent(in) :: a, b, t
658  res = erf((a-b*t)/sqrt(4*iu*b)) - erf((a+b*t)/sqrt(4*iu*b))
659  res = res * (-iu/sqrt(8*iu*b)) * exp(-iu*a**2/(4*b))
660  end function expiatbt2_intt

732  subroutine lf_get_omega(lf,zeit,omega,env)
733  ! Calculate current frequency (which changes with time when a chirp rate is specified)
734  ! returns with an error if omega is 0 or smaller
735  ! optional argument env can be given to suppress this error when env (laser field envelope) is smaller than epsilon
736  use atomic_units
737  type(laserfield), intent(in) :: lf
738  real(dp), intent(in) :: zeit
739  real(dp), intent(out) :: omega
740  real(dp), intent(inout), optional :: env
741 
742  ! The chirp rate is given in omega_0/as
743  omega = lf%omega * (1.d0 + lf%linear_chirp_rate_w0as/au_as * (zeit-lf%peak_time))
744  ! if omega is larger than or equal to zero, everything is okay and we can return
745  if (omega >= 0.d0) return
746 
747  ! omega < 0.d0 -> check envelope if present
748  if (present(env)) then
749  if (env <= epsilon(1.d0)) then
750  ! envelope <= epsilon, recover by setting envelope to ZERO
751  env = 0.d0
752  ! now we can return. we do not write error message to prevent flooding e.g. fourierlaserfield output
753  return
754  end if
755  end if
756 
757  ! omega < 0.d0 and no envelope or envelope too large
758  write(0,'(a)') ' ERROR: Field chirp is too large, omega(t) <= zero!'
759  write(0,'(a,9g15.5)') ' t, omega = ', zeit, omega
760  if (present(env)) write(0,'(a,9g15.5)') ' envelope > epsilon, env, epsilon =', env, epsilon(1.d0)
761  write(0,*) 'STOPPING!'
762  stop 519
763 
764  end subroutine lf_get_omega

766  real(dp) function lf_get_el(lf,zeit,env_out) result(EL)
767  use laserfields_miscfuncs
768  type(laserfield), intent(in) :: lf
769  real(dp), intent(in) :: zeit
770  real(dp), intent(out), optional :: env_out
771  real(dp) :: env, envpr, osc, oscpr
772  real(dp) :: omega
773 
774  if (lf%form == 'readin') then
775  if (zeit < lf%tt(1) .or. zeit > lf%tt(size(lf%tt))) then
776  el = 0.d0
777  else
778  ! interpolate a value and cycle to next laser field
779  el = interpolate(lf%tt,lf%EE,zeit,degree=6)
780  end if
781  ! set the envelope part to zero, as the readin files can not make this distinction
782  env = 0.d0
783  else
784  call lf_get_envelope(lf,zeit,env,envpr)
785 
786  if (lf%omega==0) then ! No oscillation enclosed by envelope
787  if (lf%is_vecpot) then
788  el = -envpr
789  else
790  el = env
791  end if
792  else
793  call lf_get_omega(lf,zeit,omega,env)
794  osc = sin(omega * (zeit-lf%peak_time) + pi*lf%phase_pi)
795  ! d(w(t)*(t-peak))/dt = w(t) + w'(t)*(t-peak) = omega + lf%omega * lf%chirp * (t-peak) = 2 * omega - lf%omega
796  oscpr = (2.d0*omega-lf%omega)*cos(omega*(zeit-lf%peak_time)+pi*lf%phase_pi)
797 
798  if (lf%is_vecpot) then
799  ! Divide out derivative of oscillation to ensure peak amplitude of E0
800  ! NOTE: this does not actually give a peak amplitude of E0 for chirped linear/linear2 pulses
801  el = -(env * oscpr + envpr * osc) / lf%omega
802  else ! describes electric field directly
803  el = env * osc
804  end if
805  end if
806  end if
807 
808  if (present(env_out)) env_out = env
809 
810  end function lf_get_el

976  real(dp) function laserfields_largest_possible_dt(zeit,endzeit,global) result(dt)
977  real(dp), intent(in) :: zeit
978  real(dp), intent(in), optional :: endzeit
979  real(dp) :: tstart, tend, omega, tx
980  integer :: ii
981  logical, intent(in), optional :: global
982 
983  ! if endzeit is specified, that is the time until which we want to propagate at most
984  if (present(endzeit)) then
985  if (endzeit <= zeit) then
986  ! we have already passed endzeit, but still want to do an extra step - shouldn't normally happen!
987  ! do at least a very small step - the smallest possible that still changes zeit
988  dt = nearest(zeit,1.d0)-zeit
989  return
990  end if
991  ! normally, have the time until endzeit be the maximum timestep
992  dt = endzeit-zeit
993  else
994  ! otherwise, do steps of at most 100 atomic units
995  dt = 100.d0
996  end if
997 
998  ! if global is requested, we just get the global maximum timestep
999  if (present(global)) then
1000  if (global) then
1001  dt = min(dt,minval(all_laserfields(1:n_laserfields)%TX)/minimum_steps_per_laser_period)
1002  return ! exit function
1003  end if
1004  end if
1005 
1006  ! otherwise, loop over laser fields and get the maximum timestep allowed at this time (doing at most minimum_steps_per_laser_period steps per period)
1007  do ii = 1, n_laserfields
1008  tstart = lf_get_starttime(all_laserfields(ii))
1009  tend = lf_get_endtime(all_laserfields(ii))
1010 
1011  ! this laser field is only relevant if we are before its endtime
1012  if (zeit < tend) then
1013  if (zeit >= tstart) then
1014  if (all_laserfields(ii)%omega==0) then
1015  ! for fields without oscillation, we want 500 steps for the whole field
1016  dt = min(dt,(all_laserfields(ii)%duration+2*all_laserfields(ii)%rampon)/500)
1017  else
1018  ! if it's currently active, dt has to be at most TX(zeit)/minimum_steps_per_laser_period, where TX depends on zeit for chirped pulses
1019  call lf_get_omega(all_laserfields(ii),zeit,omega)
1020  tx = twopi / omega
1021  dt = min(dt,tx/minimum_steps_per_laser_period)
1022  end if
1023  else if (zeit+dt > tstart) then
1024  ! if we would go beyond the starttime, decrease dt to go only there
1025  dt = tstart - zeit
1026  end if
1027  end if
1028  end do
1029 
1030  end function laserfields_largest_possible_dt

1059  real(dp) function laserfield_teff(lf,n_photon) result(teff)
1060  type(laserfield), intent(in) :: lf
1061  integer, intent(in) :: n_photon
1062  ! returns the "effective duration" of a laser field for n-photon processes
1063  ! the values for T_eff are calculated according to
1064  ! I_0^n * T_eff = \Int_0^T I(t)^n dt = \Int_0^T envelope(t)^(2n) dt
1065  ! calculations done in intensity_integrals.nb
1066  select case (lf%form)
1067  case('gaussianF')
1068  ! result for exp(-t^2/(r*T^2)) = T sqrt(r PI / 2n)
1069  ! for gaussian, r = 1/log(16)
1070  teff = lf%duration * sqrt(pio2/(n_photon*log(16.d0)))
1071  case('gaussianI')
1072  ! for gaussianI, r = 1/log(4)
1073  teff = lf%duration * sqrt(pio2/(n_photon*log(4.d0)))
1074  case('sin2')
1075  teff = lf%duration * gamma(0.5d0+n_photon*2) / (sqrt(pi)*gamma(1.d0+n_photon*2))
1076  case('sin_exp')
1077  teff = lf%duration * gamma(0.5d0+n_photon*lf%form_exponent) / (sqrt(pi)*gamma(1.d0+n_photon*lf%form_exponent))
1078  case('linear')
1079  teff = lf%duration + 2*lf%rampon / (1+2.d0*n_photon)
1080  case('linear2')
1081  ! flat top for lf%duration with a sin2 pulse of duration 2*lf%rampon as rampon/off
1082  teff = lf%duration + 2*lf%rampon * gamma(0.5d0+n_photon*2) / (sqrt(pi)*gamma(1.d0+n_photon*2))
1083  case default
1084  write(0,*) 'ERROR: effective duration not implemented for laser field form ', lf%form
1085  stop
1086  end select
1087  end function laserfield_teff

1151  if (lf%omega==0) then ! No oscillation enclosed by envelope
1152  elft = lf_envelope_fourier(lf,omega)
1153  else
1154  ! with tp = t-tpeak, the whole pulse is
1155  ! f(t) = env(t) sin (phi0 + w0*tp + chirp*tp**2)
1156  ! = (env(t) exp(IU*(phi0 + w0*tp + chirp*tp**2)) - c.c. ) / (2*IU)
1157  ! for the fourier transform, we include the chirp term exp(i chirp tp**2) in the envelope.
1158  ! this part is transformed in lf_envelope_fourier.
1159  ! exp(IU*phi0) is just a constant prefactor, and the linear phase w0*tp just gives a shift in frequency,
1160  ! F[f(t) exp(IU w0 t)](w) = F[f(t)](w-w0)
1161  ! complex conjugation of the transformed function gives complex conjugation + reversal of the argument in the transform, so
1162  ! F[conjg(f(t) exp(IU w0 t))](w) = conjg(F[f(t) exp(IU w0 t)](-w)) = conjg(F[f(t)](-w-w0))
1163 
1164  elft = ( lf_envelope_fourier(lf, omega - lf%omega) * exp( iu*pi*lf%phase_pi) &
1165  & - conjg(lf_envelope_fourier(lf, -omega - lf%omega)) * exp(-iu*pi*lf%phase_pi) ) / (2*iu)
1166  end if

1169  ! shift in time now -- just adds a phase exp(-IU*omega*peak_time), as F[f(t-a)] = exp(-IU*omega*a) F[f(t)]
