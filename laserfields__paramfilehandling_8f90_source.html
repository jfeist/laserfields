152  subroutine write_header(unit,name)
153  integer, parameter :: headerlen = 60
154  integer, intent(in) :: unit
155  character(len=*), intent(in) :: name

157  integer :: inlen, padl, padr
158  inlen = len_trim(name)
159  header(1:headerlen) = repeat('#',headerlen)
